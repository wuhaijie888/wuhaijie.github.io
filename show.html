高阶组件开发

	public static double JieCheng(int number)
	{
		if (number == 0)
		{
			return 0;
		}

		//初始值必须设置为1
		double result = 1; 

		for (int i = number; i >= 1; i--)
		{
			result = result*i;
		}
		return result;
	}
